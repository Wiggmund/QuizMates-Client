CREATE TABLE IF NOT EXISTS students (id BIGINT GENERATED BY DEFAULT AS IDENTITY NOT NULL,first_name VARCHAR(255) NOT NULL,last_name VARCHAR(255) NOT NULL,group_id  BIGINT REFERENCES groups (id) ON DELETE SET NULL,CONSTRAINT pk_students PRIMARY KEY (id));






CREATE TABLE IF NOT EXISTS groups (id BIGINT GENERATED BY DEFAULT AS IDENTITY NOT NULL,name VARCHAR(255) NOT NULL,CONSTRAINT pk_groups PRIMARY KEY (id),CONSTRAINT uc_groups_name UNIQUE (name));
